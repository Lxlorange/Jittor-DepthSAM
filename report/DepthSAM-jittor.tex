class DepthSAM(nn.Module):
    def __init__(self, image_encoder, decoder, prompt_encoder, mask_decoder):
        super().__init__()
        # 以npz格式加载冻结DAv2编码器
        self.image_encoder.load_state_dict("depth_anything_v2_vitl.npz")

        # 串行注入SMEA，仅gate/experts可训练
        for p in self.image_encoder.parameters(): p.requires_grad = False
        self.image_encoder.pretrained.blocks = nn.Sequential(*[
            MOEAdapter(b) for b in self.image_encoder.pretrained.blocks])
        for block in self.image_encoder.pretrained.blocks:
            for p in list(block.gate.parameters())+list(block.experts.parameters()):
                p.requires_grad = True        

        for m in [prompt_encoder.point_embeddings, prompt_encoder.pe_layer_2,
                  prompt_encoder.not_a_point_embed, prompt_encoder.no_mask_embed]:
            for p in m.parameters(): p.requires_grad = False

    def execute(self, x, batched_input):
        # DAv2提多尺度特征 → GSFM融合 → SAM解码器输出最终掩码
        depth, features = self.image_encoder(x)
        out, embeddings = self.decoder(features[3], features[2], features[1], features[0])
        outputs = []
        for record, emb, hint in zip(batched_input, embeddings, out):
            self.prompt_encoder.image_embedding_size = tuple(emb.shape[-2:])
            sparse, dense = self.prompt_encoder(points=None, boxes=record.get("boxes"), masks=hint.unsqueeze(0))
            low_res, _ = self.mask_decoder(emb.unsqueeze(0), self.prompt_encoder.get_dense_pe(), sparse, dense)
            outputs.append(low_res)
        return jt.concat(outputs, dim=0)
\end{lstlisting}

class MOEAdapter(nn.Module):
    def __init__(self, blk, num_experts=8, top_k=4):
        super().__init__()

        # 保存原始 ViT Transformer block；该 block 本身保持冻结
        self.block = blk

        # N 个专家；实验中可通过 DEPTHSAM_MOE_EXPERTS 调整专家数
        self.num_experts = max(1, int(os.environ.get(
            "DEPTHSAM_MOE_EXPERTS", num_experts)))
        self.top_k = min(top_k, self.num_experts)
        self.current_aux_loss = jt.array(0.0)

        # token hidden dim 来自原始 block 的 qkv 输入维度
        dim = blk.attn.qkv.in_features

        # G(x)：门控网络，为每个 token 预测 expert 权重
        self.gate = nn.Linear(dim, self.num_experts)

        # E_i(x)：轻量 MLP expert，瓶颈维度为 32
        self.experts = nn.ModuleList([
            nn.Sequential(
                nn.Linear(dim, 32),
                nn.GELU(),
                nn.Linear(32, dim),
                nn.GELU(),
            )
            for _ in range(self.num_experts)
        ])
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{MOEAdapter}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def execute(self, x):
    # x: (B, N, C)，N 是 token 数
    B, N, C = x.shape
    x_flat = x.reshape(B * N, C)

    # 1. 计算全部 expert 的门控概率 G(x)
    gate_logits = self.gate(x_flat)
    gate_weights = nn.softmax(gate_logits, dim=-1)

    # 2. 退化情形：单 expert 时不再需要 top-k routing
    if self.num_experts == 1:
        routing_weights = jt.ones_like(gate_weights)
        self.current_aux_loss = gate_weights.sum() * 0.0
    else:
        # 3. 保留 Top-k experts，并归一化其权重
        top_w, top_idx = self._topk(gate_weights)
        top_w = top_w / (jt.sum(top_w, dims=[-1], keepdims=True) + 1e-8)

        # 4. 构造稀疏 routing 权重：非 Top-k expert 权重为 0
        routing_weights = jt.zeros_like(gate_weights)
        expert_ids = jt.arange(self.num_experts).reshape((1, self.num_experts))
        for i in range(self.top_k):
            idx = top_idx[:, i].reshape((-1, 1))
            w = top_w[:, i].reshape((-1, 1))
            routing_weights += (idx == expert_ids).float32() * w

        # 5. MoE 负载均衡辅助损失：alpha * N * sum_i f_i P_i
        prob = jt.mean(gate_weights, dims=[0])
        mask = routing_weights > 0
        frac = jt.sum(mask.float32(), dims=[0]) / (B * N + 1e-8)
        self.current_aux_loss = jt.sum(prob * frac) * self.num_experts
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{MOEAdapter}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def execute(self, x):
    ...
    # 6. 为保证 Jittor 图稳定，先计算所有 expert，再按 routing 权重聚合
    expert_outputs = []
    for e_idx in range(self.num_experts):
        expert_outputs.append(self.experts[e_idx](x_flat))
    expert_outputs = jt.stack(expert_outputs, dim=1)

    # 7. 加权求和得到 MoE 调制量
    output = jt.sum(
        expert_outputs * routing_weights.unsqueeze(-1),
        dims=[1]
    ).reshape(B, N, C)

    # 8. 残差注入：x_m = x + sum_i(g_i E_i(x)) / sum_i(g_i)
    prompted = x + output

    # 9. 调制后的 token 再进入冻结的原始 Transformer block
    return self.block(prompted)

class MOELossCollector:
    def get_total_loss(self, model):
        # 遍历所有 MOEAdapter，收集每层 current_aux_loss
        total = None
        for module in model.image_encoder.pretrained.blocks:
            if isinstance(module, MOEAdapter):
                aux = module.current_aux_loss
                total = aux if total is None else total + aux
        return jt.array(0.0) if total is None else total * self.alpha

class DepthAnythingV2(nn.Module):
    def __init__(self, encoder="vitl", features=256,
                 out_channels=[256, 512, 1024, 1024], ...):
        super().__init__()
        # 不同规模 DINOv2 对应不同中间层；本项目使用 vitl
        self.intermediate_layer_idx = {
            "vitl": [4, 11, 17, 23],
            "vitg": [9, 19, 29, 39],
        }
        self.encoder = encoder
        # pretrained 对应冻结的 DINOv2 backbone
        self.pretrained = DINOv2(model_name=encoder)
        # DPTHead: ViT tokens -> dense depth + multi-scale features
        self.depth_head = DPTHead(self.pretrained.embed_dim, features,
                                  out_channels=out_channels)

    def execute(self, x):
        # DINOv2 patch size = 14，H/W 需要能被 14 整除
        patch_h, patch_w = x.shape[-2] // 14, x.shape[-1] // 14
        # 取 4 个中间层 token，保留 cls token
        feats = self.pretrained.get_intermediate_layers(
            x, self.intermediate_layer_idx[self.encoder],
            return_class_token=True)
        # depth 用于几何监督/深度输出，fts 送入后续 GSFM
        depth, fts = self.depth_head(feats, patch_h, patch_w)
        return nn.relu(depth), fts

\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class DPTHead(nn.Module):
    def __init__(self, in_channels, features=256,
                 out_channels=[256, 512, 1024, 1024], ...):
        super().__init__()

        # 1x1 Conv：把四层 DINOv2 token 投影到 DPT 通道数
        self.projects = nn.ModuleList([
            nn.Conv2d(in_channels, c, kernel_size=1)
            for c in out_channels
        ])

        # 四个尺度对齐到 DPT decoder 需要的空间分辨率
        self.resize_layers = nn.ModuleList([
            nn.ConvTranspose2d(out_channels[0], out_channels[0], 4, 4),
            nn.ConvTranspose2d(out_channels[1], out_channels[1], 2, 2),
            nn.Identity(),
            nn.Conv2d(out_channels[3], out_channels[3], 3, stride=2, padding=1),
        ])

        # scratch 中包含 layer*_rn 和 refinenet*，对应 DPT decoder
        self.scratch = _make_scratch(out_channels, features)
        self.scratch.refinenet4 = _make_fusion_block(features, use_bn=False)
        self.scratch.refinenet3 = _make_fusion_block(features, use_bn=False)
        self.scratch.refinenet2 = _make_fusion_block(features, use_bn=False)
        self.scratch.refinenet1 = _make_fusion_block(features, use_bn=False)

        # 输出深度图的预测头；中间 path_4...path_1 还会作为 GSFM 输入
        self.scratch.output_conv1 = nn.Conv2d(features, features // 2, 3, padding=1)
        self.scratch.output_conv2 = nn.Sequential(
            nn.Conv2d(features // 2, 32, 3, padding=1),
            nn.ReLU(), nn.Conv2d(32, 1, 1), nn.ReLU())
\end{lstlisting}

\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def execute(self, out_features, patch_h, patch_w):
    out = []
    for i, x in enumerate(out_features):
        # patch tokens: (B,N,C) -> (B,C,H,W)
        x = x[0]
        x = x.permute(0, 2, 1).reshape((x.shape[0], x.shape[-1], patch_h, patch_w))
        # 投影通道并调整空间尺度
        x = self.projects[i](x)
        x = self.resize_layers[i](x)
        out.append(x)

    layer_1, layer_2, layer_3, layer_4 = out
    # DPT 自顶向下融合：深层语义逐步融合浅层细节
    l1, l2 = self.scratch.layer1_rn(layer_1), self.scratch.layer2_rn(layer_2)
    l3, l4 = self.scratch.layer3_rn(layer_3), self.scratch.layer4_rn(layer_4)
    path_4 = self.scratch.refinenet4(l4, size=l3.shape[2:])
    path_3 = self.scratch.refinenet3(path_4, l3, size=l2.shape[2:])
    path_2 = self.scratch.refinenet2(path_3, l2, size=l1.shape[2:])
    path_1 = self.scratch.refinenet1(path_2, l1)

    # features = [F4,F3,F2,F1]，后续送入 Decode / GSFM
    features = [path_4, path_3, path_2, path_1]
    depth = self.scratch.output_conv2(
        nn.interpolate(self.scratch.output_conv1(path_1),
                       (patch_h * 14, patch_w * 14),
                       mode="bilinear", align_corners=True))
    return depth, features
\end{lstlisting}
\end{frame}

def infer_image(self, raw_image, input_size=518):
    # 推理接口：OpenCV 图像 -> 网络张量 -> 原图尺寸深度图
    image, (h, w) = self.image2tensor(raw_image, input_size)
    depth, _ = self.execute(image)
    depth = nn.interpolate(depth, (h, w),
                           mode="bilinear", align_corners=True)[0, 0]
    return depth.numpy()

def image2tensor(self, raw_image, input_size=518):
    # 保持比例，且保证输入尺寸是 patch size 14 的倍数
    transform = Compose([
        Resize(width=input_size, height=input_size,
               resize_target=False,
               keep_aspect_ratio=True,
               ensure_multiple_of=14,
               resize_method="lower_bound",
               image_interpolation_method=cv2.INTER_CUBIC),
        NormalizeImage(mean=[0.485, 0.456, 0.406],
                       std=[0.229, 0.224, 0.225]),
        PrepareForNet(),
    ])

    h, w = raw_image.shape[:2]
    image = cv2.cvtColor(raw_image, cv2.COLOR_BGR2RGB) / 255.0
    image = transform({"image": image})["image"]
    return jt.array(image).unsqueeze(0), (h, w)

class Resize(object):
    def __call__(self, sample):
        # 按 lower_bound / keep_aspect_ratio 计算新尺寸
        # constrain_to_multiple_of 保证 H,W 是 14 的倍数
        width, height = self.get_size(
            sample["image"].shape[1],
            sample["image"].shape[0])
        sample["image"] = cv2.resize(
            sample["image"], (width, height),
            interpolation=self.__image_interpolation_method)
        return sample

class NormalizeImage(object):
    def __init__(self, mean, std):
        self.__mean = mean
        self.__std = std

    def __call__(self, sample):
        # ImageNet mean/std，与 DINOv2 预训练输入分布对齐
        sample["image"] = (sample["image"] - self.__mean) / self.__std
        return sample

class PrepareForNet(object):
    def __call__(self, sample):
        # HWC -> CHW，并转成连续 float32，方便 Jittor 构图
        image = np.transpose(sample["image"], (2, 0, 1))
        sample["image"] = np.ascontiguousarray(image).astype(np.float32)
        return sample
\end{lstlisting}

\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class Decode(nn.Module):
    def __init__(self, in1, in2, in3, in4):
        super().__init__()
        # premp: 最终融合特征 -> 粗 mask prompt
        self.premp = nn.Sequential(
            nn.ConvTranspose2d(in1, in1 // 2, kernel_size=2, stride=2),
            LayerNorm2d(in1 // 2), nn.GELU(),
            nn.Conv2d(in1 // 2, 1, kernel_size=3, padding=1),
        )
        # 三阶段渐进融合；fm1 使用轻量路径以控制 Jittor 显存
        self.fm1 = MEF(in1, in2, use_attention=False)
        self.fm2 = MEF(in2, in3)

class PositionEmbeddingRandom(nn.Module):
    def __init__(self, num_pos_feats=64, scale=1.0):
        super().__init__()
        # 随机高斯矩阵，将二维坐标映射到高频位置特征
        self.register_buffer(
            "positional_encoding_gaussian_matrix",
            scale * jt.randn((2, num_pos_feats)),
        )

    def _pe_encoding(self, coords):
        # coords 归一化到 [0,1] 后映射为 sin/cos 编码
        coords = 2 * coords - 1
        coords = coords @ self.positional_encoding_gaussian_matrix
        coords = 2 * np.pi * coords
        return jt.concat([jt.sin(coords), jt.cos(coords)], dim=-1)

    def execute(self, size):
        # 为 HxW image embedding 生成 dense positional encoding
        h, w = size
        grid = jt.ones((h, w), dtype=jt.float32)
        y_embed = (grid.cumsum(dim=0) - 0.5) / h
        x_embed = (grid.cumsum(dim=1) - 0.5) / w
        pe = self._pe_encoding(jt.stack([x_embed, y_embed], dim=-1))
        return pe.permute(2, 0, 1)

    def forward_with_coords(self, coords, image_size):
        # 点/框坐标先除以原图尺寸，再做同样的位置编码
        coords = coords.clone()
        coords[:, :, 0] = coords[:, :, 0] / image_size[1]
        coords[:, :, 1] = coords[:, :, 1] / image_size[0]
        return self._pe_encoding(coords.float32())
\end{lstlisting}

class Attention(nn.Module):
    def __init__(self, embedding_dim, num_heads, downsample_rate=1):
        super().__init__()
        self.internal_dim = embedding_dim // downsample_rate
        self.num_heads = num_heads
        self.q_proj = nn.Linear(embedding_dim, self.internal_dim)
        self.k_proj = nn.Linear(embedding_dim, self.internal_dim)
        self.v_proj = nn.Linear(embedding_dim, self.internal_dim)
        self.out_proj = nn.Linear(self.internal_dim, embedding_dim)

    def _separate_heads(self, x, num_heads):
        b, n, c = x.shape
        x = x.reshape(b, n, num_heads, c // num_heads)
        return x.transpose(1, 2)

    def execute(self, q, k, v):
        q, k, v = self.q_proj(q), self.k_proj(k), self.v_proj(v)
        q = self._separate_heads(q, self.num_heads)
        k = self._separate_heads(k, self.num_heads)
        v = self._separate_heads(v, self.num_heads)
        _, _, _, c_per_head = q.shape
        attn = (q @ k.permute(0, 1, 3, 2)) / math.sqrt(c_per_head)
        out = nn.softmax(attn, dim=-1) @ v
        out = self._recombine_heads(out)
        return self.out_proj(out)

\begin{frame}{训练目标}
  \small
  训练阶段保留 COD 分割常用的结构损失，并叠加 SMEA 的专家负载均衡辅助项：

  \vspace{0.5em}
  \[
    \mathcal{L}
    =
    \mathcal{L}_{wBCE}
    +
    \mathcal{L}_{wIoU}
    +
    \alpha N \sum_{i=1}^{N} f_i P_i
  \]

  \vspace{0.5em}
  \begin{itemize}
    \item $\mathcal{L}_{wBCE}$：边界加权 BCE，用于像素级前景/背景监督。
    \item $\mathcal{L}_{wIoU}$：边界加权 IoU，用于优化整体区域重叠。
    \item $\alpha N \sum_i f_iP_i$：MoE load-balance loss，避免 token 长期集中到少数 experts。
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Structure Loss}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def structure_loss(pred, mask, eps=1e-8):
    # 1. 边界权重：mask 与局部平均差异越大，权重越高
    weit = 1 + 5 * jt.abs(
        nn.avg_pool2d(mask, kernel_size=31, stride=1, padding=15) - mask)

    # 2. stable BCE with logits，避免 exp 溢出导致 NaN
    max_val = jt.clamp(-pred, min_v=0)
    log_term = max_val + ((-max_val).exp() + (-pred - max_val).exp()).log()
    wbce = (1 - mask) * pred + log_term
    wbce = jt.sum(weit * wbce, dims=[2, 3]) / (jt.sum(weit, dims=[2, 3]) + eps)

    # 3. weighted IoU：同样使用边界权重
    pred = jt.sigmoid(pred)
    inter = jt.sum((pred * mask) * weit, dims=[2, 3])
    union = jt.sum((pred + mask) * weit, dims=[2, 3])
    wiou = 1 - (inter + 1) / (union - inter + 1 + eps)

    return (wbce + wiou).mean()
\end{lstlisting}

                aux = module.current_aux_loss
                total = aux if total is None else total + aux
        return jt.array(0.0) if total is None else total * self.alpha

# train.py 中最终优化目标
loss_main = structure_loss(pred, gt)
loss_aux = moe_loss_collector.get_total_loss(generator)
loss = loss_main + loss_aux
optimizer.step(loss)
\end{lstlisting}
\end{frame}
