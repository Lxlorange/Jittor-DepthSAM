class DepthSAM(nn.Module):
    def __init__(self, image_encoder, decoder, prompt_encoder, mask_decoder):
        super().__init__()
        # 以npz格式加载冻结DAv2编码器
        self.image_encoder.load_state_dict("depth_anything_v2_vitl.npz")

        # 串行注入SMEA，仅gate/experts可训练
        for p in self.image_encoder.parameters(): p.requires_grad = False
        self.image_encoder.pretrained.blocks = nn.Sequential(*[
            MOEAdapter(b) for b in self.image_encoder.pretrained.blocks])
        for block in self.image_encoder.pretrained.blocks:
            for p in list(block.gate.parameters())+list(block.experts.parameters()):
                p.requires_grad = True        

        for m in [prompt_encoder.point_embeddings, prompt_encoder.pe_layer_2,
                  prompt_encoder.not_a_point_embed, prompt_encoder.no_mask_embed]:
            for p in m.parameters(): p.requires_grad = False

    def execute(self, x, batched_input):
        # DAv2提多尺度特征 → GSFM融合 → SAM解码器输出最终掩码
        depth, features = self.image_encoder(x)
        out, embeddings = self.decoder(features[3], features[2], features[1], features[0])
        outputs = []
        for record, emb, hint in zip(batched_input, embeddings, out):
            self.prompt_encoder.image_embedding_size = tuple(emb.shape[-2:])
            sparse, dense = self.prompt_encoder(points=None, boxes=record.get("boxes"), masks=hint.unsqueeze(0))
            low_res, _ = self.mask_decoder(emb.unsqueeze(0), self.prompt_encoder.get_dense_pe(), sparse, dense)
            outputs.append(low_res)
        return jt.concat(outputs, dim=0)
\end{lstlisting}

\begin{frame}[fragile]{SMEA / MOEAdapter}
\[
  x_m = x + \frac{\sum_{i=1}^{N} g_i E_i(x)}{\sum_{i=1}^{N} g_i}
\]
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class MOEAdapter(nn.Module):
    def __init__(self, frozen_block, num_experts=8, top_k=4):
        # block 是冻结的原始 DINOv2 Transformer block
        self.block = frozen_block
        self.top_k = top_k
        dim = frozen_block.attn.qkv.in_features

        # G(x)：门控网络；E_i(x)：MLP experts
        self.gate = nn.Linear(dim, num_experts)
        self.experts = nn.ModuleList([
            nn.Sequential(nn.Linear(dim, 32), nn.GELU(),
                          nn.Linear(32, dim), nn.GELU())
            for _ in range(num_experts)
        ])

    def execute(self, x):
        gate = nn.softmax(self.gate(flatten_tokens(x)), dim=-1)
        routing = topk_and_normalize(gate, k=self.top_k)
        moe = weighted_sum([E(flatten_tokens(x)) for E in self.experts], routing)
        return self.block(x + restore_shape(moe, x))
\end{lstlisting}
\end{frame}

for e in experts:
    mask = (top_idx == e)
    output[mask] += top_w[mask] * expert_e(x_flat[mask])

# Jittor: dense experts + routing mask，避免动态分组和原地写回
expert_outputs = jt.stack([expert(x_flat) for expert in experts], dim=1)
output = jt.sum(expert_outputs * routing_weights.unsqueeze(-1), dims=1)

# 负载均衡：避免gate长期只选择少数专家
P_i = jt.mean(gate_weights, dims=0)                 # 平均门控概率
f_i = jt.mean((routing_weights > 0).float32(), dims=0)  # 实际选中比例
aux_loss = jt.sum(P_i * f_i) * num_experts

\begin{frame}[fragile]{DepthAnythingV2}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class DepthAnythingV2(nn.Module):
    def __init__(self, encoder="vitl"):
        # Frozen DINOv2 backbone；DepthSAM使用ViT-L
        self.encoder = encoder
        self.pretrained = DINOv2(model_name=encoder)

        # 从DINOv2不同深度取四层token，作为DPT decoder输入
        self.intermediate_layer_idx = {"vitl": [4, 11, 17, 23]}

        # DPTHead把ViT token恢复为空间特征，并输出深度图
        self.depth_head = DPTHead(self.pretrained.embed_dim,
                                  out_channels=[256, 512, 1024, 1024])

    def execute(self, x):
        # DINOv2 patch size = 14，因此输入尺寸需要能被14整除
        patch_h, patch_w = x.shape[-2] // 14, x.shape[-1] // 14

        # get_intermediate_layers返回4层中间token和cls token
        tokens = self.pretrained.get_intermediate_layers(
            x, self.intermediate_layer_idx[self.encoder], return_class_token=True)

        # depth用于深度估计输出；features送入后续GSFM
        depth, features = self.depth_head(tokens, patch_h, patch_w)
        return nn.relu(depth), features
\end{lstlisting}

\begin{frame}[fragile]{DPTHead}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class DPTHead(nn.Module):
    def __init__(self, in_channels, features=256, out_channels=[256, 512, 1024, 1024]):
        # 1. project：四层token分别用1x1卷积投影到DPT通道数
        self.projects = nn.ModuleList([ nn.Conv2d(in_channels, c, kernel_size=1) for c in out_channels])

        # 2. resize：把四层特征调整到DPT需要的多尺度金字塔
        self.resize_layers = nn.ModuleList([
            nn.ConvTranspose2d(out_channels[0], out_channels[0], 4, 4),
            nn.ConvTranspose2d(out_channels[1], out_channels[1], 2, 2),
            nn.Identity(),
            nn.Conv2d(out_channels[3], out_channels[3], 3, stride=2, padding=1),])

        # 3. scratch/refinenet：DPT自顶向下融合，恢复空间细节
        self.scratch = _make_scratch(out_channels, features)
        self.refinenets = [refinenet4, refinenet3, refinenet2, refinenet1]

        # 4. depth head：从最高分辨率特征预测单通道深度图
        self.output_head = nn.Sequential(
            nn.Conv2d(features, features // 2, 3, padding=1),
            nn.Upsample(scale_factor=14, mode="bilinear"),
            nn.Conv2d(features // 2, 32, 3, padding=1),
            nn.ReLU(), nn.Conv2d(32, 1, 1), nn.ReLU(),)

    def execute(self, tokens, patch_h, patch_w):
        # token序列 -> 2D feature map -> project/resize -> DPT融合
        maps = [token_to_feature_map(t, patch_h, patch_w) for t in tokens]
        maps = [project_and_resize(m) for m in maps]
        F4, F3, F2, F1 = top_down_refinenet(maps)
        depth = depth_prediction_head(F1)
        return depth, [F4, F3, F2, F1]
\end{lstlisting}
\end{frame}

class Decode(nn.Module):
    def __init__(self):
        # 三阶段渐进融合：深层语义 -> 浅层细节
        self.fm3 = MEF(F2_channels, F1_channels)
        self.fm2 = MEF(F3_channels, F2_channels)
        self.fm1 = MEF(F4_channels, F3_channels, use_attention=False)

        # 输出粗 mask，作为 SAM 的 mask prompt
        self.premp = nn.Sequential(ConvTranspose2d(...), LayerNorm2d(...),
                                  GELU(), Conv2d(..., out_channels=1))

    def execute(self, F4, F3, F2, F1):
        br3 = self.fm3(F2, F1)
        br2 = self.fm2(F3, br3)
        br1 = self.fm1(F4, br2)
        coarse_mask = self.premp(br1)
        return coarse_mask, br1

class MEF(nn.Module):
    def execute(self, current, deeper):
        # 对齐尺度后拼接，再送入 FM/SFRM
        deeper = ConvTranspose2d(...)(deeper)
        return self.fm(jt.concat([current, deeper], dim=1))

class FM(nn.Module):
    def __init__(self, dim, out_dim, use_attention=True):
        # 先压缩通道，再做双域注意力与局部增强
        self.reduce = BasicConv2d(dim, out_dim, kernel_size=1)
        self.sfrm = Attention_SD(out_dim) if use_attention else Identity()
        self.local = DWConv(out_dim)   # 只作为局部增强，不展开讲

    def execute(self, x):
        x = self.reduce(x)
        x = self.sfrm(x, x)
        return self.local(x)
\end{lstlisting}

\begin{frame}[fragile]{SFRM / Attention\_SD}
\[
\begin{aligned}
F_F &= \mathcal{F}^{-1}(\operatorname{Softmax}(QK^T)V),\quad
F_S = \operatorname{Softmax}(qk^T)v \\
F_{S\to F} &= \mathcal{F}^{-1}(\mathcal{F}(\operatorname{Softmax}(qk^T))V),\quad
F_{F\to S}=\mathcal{F}^{-1}(\operatorname{Softmax}(QK^T))v
\end{aligned}
\]
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def execute(self, x, d):
    # 频域分支：Jittor ComplexNumber + _fft2/_ifft2
    Q, K, V = _fft2(qr(x)), _fft2(kr(x)), _fft2(vr(x))
    attn_f = complex_softmax(normalize(Q) @ normalize(K).T)
    F_F = abs(_ifft2(attn_f @ V))

    # 空间分支：普通 attention
    q, k, v = qm(d), km(d), vm(d)
    attn_s = softmax(normalize(q) @ normalize(k).T)
    F_S = attn_s @ v

    # 交叉引导：空间 attention 作用于频域 value；频域 attention 作用于空间 value
    F_S2F = abs(_ifft2(_fft2(attn_s) @ V))
    F_F2S = attn_f.real @ v
    return project(concat([F_S2F, F_F, F_F2S, F_S]))
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Jittor FFT / IFFT封装}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
# Jittor 没有像 torch.fft.fft2 这样直接作用在实数 Var 上的接口。
# Attention_SD 中需要频域注意力，因此把实数特征封装为 ComplexNumber 再调用 fft2/ifft2。

def _complex(x):
    # 实数 Var -> 复数类；虚部补 0
    if isinstance(x, nn.ComplexNumber):
        return x
    return nn.ComplexNumber(x, jt.zeros_like(x))

def _fft2(x):
    x = _complex(x)
    shape = x.real.shape
    # ComplexNumber.fft2 只处理最后两个维度；先把 batch/channel/head 展平
    flat = nn.ComplexNumber(
        x.real.reshape((-1, shape[-2], shape[-1])),
        x.imag.reshape((-1, shape[-2], shape[-1])),
    )
    out = flat.fft2()
    # 再恢复到原始形状，供后续 attention 计算
    return nn.ComplexNumber(out.real.reshape(shape), out.imag.reshape(shape))

def _ifft2(x):
    x = _complex(x)
    shape = x.real.shape
    flat = nn.ComplexNumber(
        x.real.reshape((-1, shape[-2], shape[-1])),
        x.imag.reshape((-1, shape[-2], shape[-1])),
    )
    out = flat.ifft2()
    return nn.ComplexNumber(out.real.reshape(shape), out.imag.reshape(shape))
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{PromptEncoder}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class PromptEncoder(nn.Module):
    def __init__(self, embed_dim=256, mask_in_chans=16, ...):
        # SAM原有的点/框prompt embedding保留，但DepthSAM主要不用
        self.point_embeddings = nn.ModuleList([...])
        self.not_a_point_embed = nn.Embedding(1, embed_dim)

        # 核心：把GSFM输出的coarse mask编码为dense prompt
        self.mask_downscaling = nn.Sequential(
            nn.Conv2d(1, mask_in_chans // 4, kernel_size=2, stride=2),
            LayerNorm2d(mask_in_chans // 4), nn.GELU(),
            nn.Conv2d(mask_in_chans // 4, mask_in_chans, kernel_size=2, stride=2),
            LayerNorm2d(mask_in_chans), nn.GELU(),
            nn.Conv2d(mask_in_chans, embed_dim, kernel_size=1),
        )
        self.pe_layer = PositionEmbeddingRandom(embed_dim // 2)

    def execute(self, points=None, boxes=None, masks=None):
        # sparse: 点/框prompt；本项目通常为空
        sparse = jt.empty((batch_size(points, boxes, masks), 0, self.embed_dim))

        # dense: coarse mask prompt -> image embedding同尺度的dense embedding
        masks = nn.interpolate(masks, self.mask_input_size, mode="bilinear")
        dense = self.mask_downscaling(masks)
        return sparse, dense

    def get_dense_pe(self):
        # 位置编码送入MaskDecoder的TwoWayTransformer
        return self.pe_layer(self.image_embedding_size).unsqueeze(0)
\end{lstlisting}
\end{frame}

class MaskDecoder(nn.Module):
    def __init__(self, transformer_dim=256, transformer=TwoWayTransformer(...)):
        # iou token预测质量；mask tokens生成候选mask
        self.iou_token = nn.Embedding(1, transformer_dim)
        self.mask_tokens = nn.Embedding(num_mask_tokens, transformer_dim)
        self.transformer = transformer

        # transformer输出的image feature上采样
        self.output_upscaling = nn.Sequential(
            nn.ConvTranspose2d(transformer_dim, transformer_dim // 4, 2, 2),
            LayerNorm2d(transformer_dim // 4), nn.GELU(),
            nn.ConvTranspose2d(transformer_dim // 4, transformer_dim // 8, 2, 2),
        )
        # 每个mask token经过MLP生成一组动态mask权重
        self.output_hypernetworks_mlps = nn.ModuleList([MLP(...) for _ in mask_tokens])

    def predict_masks(self, image_embedding, image_pe, sparse, dense):
        tokens = concat([iou_token, mask_tokens, sparse])
        src = image_embedding + dense
        hs, src = self.transformer(src, image_pe, tokens)

        up = self.output_upscaling(src)
        hyper = stack([MLP(mask_token) for mask_token in hs.mask_tokens])
        masks = hyper @ flatten(up)
        return masks, iou_prediction(hs.iou_token)
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{TwoWayTransformer}
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
class TwoWayAttentionBlock(nn.Module):
    def execute(self, prompt_tokens, image_tokens, prompt_pe, image_pe):
        # 1. prompt tokens 内部自注意力
        prompt_tokens = self_attn(prompt_tokens + prompt_pe)

        # 2. prompt -> image：prompt 读取图像信息
        prompt_tokens = cross_attn(q=prompt_tokens + prompt_pe,
                                   k=image_tokens + image_pe,
                                   v=image_tokens)

        # 3. prompt tokens MLP
        prompt_tokens = prompt_tokens + mlp(prompt_tokens)

        # 4. image -> prompt：图像 token 反向接收 prompt 信息
        image_tokens = cross_attn(q=image_tokens + image_pe,
                                  k=prompt_tokens + prompt_pe,
                                  v=prompt_tokens)
        return prompt_tokens, image_tokens
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Loss 实现}
Loss由结构损失和 SMEA 的专家负载均衡项组成：
\[
  \mathcal{L} = \mathcal{L}_{wBCE} + \mathcal{L}_{wIoU} + \alpha N \sum_{i=1}^{N} f_i P_i
\]
\begin{lstlisting}[style=pythoncode,basicstyle=\ttfamily\tiny]
def structure_loss(pred, mask, eps=1e-8):
    # 边界权重：mask与局部均值差异越大，权重越高
    weit = 1 + 5 * abs(avg_pool(mask, k=31) - mask)

    # Jittor端使用 stable BCE with logits，避免 exp/log 造成 NaN
    wbce = stable_bce_with_logits(pred, mask)
    wbce = sum(weit * wbce) / (sum(weit) + eps)

    pred = sigmoid(pred)
    inter = sum((pred * mask) * weit)
    union = sum((pred + mask) * weit)
    wiou = 1 - (inter + 1) / (union - inter + 1 + eps)
    return mean(wbce + wiou)

# 训练时叠加 MoE aux loss
loss_main = structure_loss(pred, gt)
loss_aux = moe_loss_collector.get_total_loss(generator)
loss = loss_main + loss_aux
\end{lstlisting}
\end{frame}
